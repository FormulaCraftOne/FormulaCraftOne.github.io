$$ \begin{align}
f(t) &= \int_{-\infty}^{\infty}{F(\omega)e^{2\pi i \omega t}}d\omega \\
F(\omega) &= \int_{-\infty}^{\infty}{f(t)e^{-2\pi i \omega t}}dt
\end{align}
